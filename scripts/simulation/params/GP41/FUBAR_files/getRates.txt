hi<-read.csv('tree.tre.fubar.csv')
w<-hi$beta/hi$alpha
w<-w[w<=6]
counts<-hist(w, breaks=50)$counts # This won't return 50 categories (58!) but if you try a lower number it returns like 30 categories. Doesn't really matter as long as we have substantial variation.
probs<-counts/length(w)
categories<-hist(w, breaks=50)$breaks
cat<-categories[2:length(categories)] ## since includes starting bin value.
cbind(cat, probs)
